\begin{algorithm}[H]
\caption{MaterialGeneration}
\label{alg:material-expansion}
\begin{algorithmic}[1]
\REQUIRE Current state $S$
\FOR{$i = 1$ \TO $S$.$configuration$.$maxMaterialIterations$}
    \STATE $prompt$ \gets \text{$S$.$configuration$.$materials$} \cup \text{$S$.$materials$}$
    \STATE  $R$ \gets \CALL{MaterialDiscoveryLLMAgent}{($prompt$)}
    \STATE Filter/Merge duplicates in $R$ by HS code and aliases
    \FOR{each $M$ in $R$}
        \IF{$M \notin \text{$S.$materials}$}
            \STATE \text{$S$.$materials$} \gets \text{$S$.$materials$} \cup \text{$M$}
        \ELSE
            \STATE Merge $M.name$ and $M.aliases$ into $S.materials$ as aliases
        \ENDIF
    \ENDFOR
\ENDFOR
\end{algorithmic}
\end{algorithm}



\begin{algorithm}[H]
\caption{ProcessGeneration}
\label{alg:process-expansion}
\begin{algorithmic}[1]
\REQUIRE Current state $S$, options $O$ (from state)
\FOR{$j = 1$ to $O.process\_passes$}
    \FOR{each $M$ in $\text{$S$.$material$}$}
        \STATE \textbf{Mode} \gets \text{'create'}
        \STATE $\mathcal{P} \gets \CALL{GetMaterialProcesses}{($M$,$S$})
        \STATE $AllScored$ \gets \text{True if len($\mathcal{P}$) > 0 and all $\mathcal{P}$ have been scored}$
        \IF{len($\mathcal{P}) > 0$ \AND $AllScored$}
            \STATE \textbf{continue}
        \ENDIF
        \IF{len($\mathcal{P})> 0$}
            \STATE \textbf{Mode} \gets \text{'edit'}
            \FOR{each $p$ in $\mathcal{P}$}
                \STATE \text{$topRefs$,$allReviewed$,$foundRequiredDistinctDomains$,$unreviewedCount$} \gets \CALL{EvaluateReferences}{(\text{$p$.$references$}},\text{$O$.$minimumProcessReferences$},\text{$O$.$distinctDomainsRequired$)}}
                \STATE $prompt$ \gets \text{$prompt$} \cup \text{$p$}
            \ENDFOR
             \STATE $prompt$ \gets \text{$prompt$} \cup \text{Edit Mode Instructions}$
        \ELSE
            \STATE $prompt$ \gets \text{$prompt$} \cup \text{Create Mode Instructions}
        \ENDIF
        \STATE  $R$ \gets \CALL{ProcessDiscoveryLLMAgent}{($prompt$)}$
        \FOR{each $instruction$ in $R$}
            \STATE \CALL{ProcessInstruction}{(\text{$instruction$}, \text{$S$})} \COMMENT{Adds/Modifies/removes processes in the process list}
        \ENDFOR
        \STATE Prompt LLM to generate candidate processes for $m$
        \STATE Add new processes to $\mathcal{E}$
    \ENDFOR
    \STATE Merge duplicate processes
    \STATE \CALL{ReviewProcesses}{$\mathcal{E}, \mathcal{R}$}
\ENDFOR
\end{algorithmic}
\end{algorithm}


\begin{algorithm}
\caption{ReviewProcesses}
\label{alg:review-processes}
\begin{algorithmic}[1]
\REQUIRE Current state $S$, options $O$ (from state)
\FOR{each process $p \in S.processes$}
    \STATE $refReplacements$ \gets \emptyset
    \FOR{each reference $r \in p.references$}
        \IF{$r.source\_quality\_score = \text{None}$}
            \STATE $r.source\_quality\_score$,$r.source\_quality\_reason$ \gets$ \CALL{URLScorerLLMAgent}$(r$.url)$
        \ENDIF
        \IF{$r.content\_quality\_score = \text{None}$}
            \STATE $content \gets  \text{Fetch content from $r.url$ as markdown}$}
            \IF{$content = \text{error}$}
                \STATE $r.content\_quality\_score \gets 0.0$
                \STATE $r.content\_quality\_reason \gets \text{error message}$
            \ELSE
                \STATE $review \gets$ \CALL{URLScorerLLMAgent}$($content$)$ 
                \IF{$review \ne \text{None}$}
                    \STATE $r.content\_quality\_score \gets review.score$
                    \STATE $r.content\_quality\_reason \gets review.reason$
                \ENDIF
            \ENDIF
        \ELSE
        
        \ENDIF
        \STATE Append $r$ to $refReplacements$
    \ENDFOR
    \STATE \text{$topRefs$,$allReviewed$,$foundRequiredDistinctDomains$,$unreviewedCount$} \gets \CALL{EvaluateReferences}{(\text{$p$.$references$}},\text{$O$.$minimumProcessReferences$},\text{$O$.$distinctDomainsRequired$)}}
    \STATE $process\_score \gets 0.0$
    \FORALL{ref $\in$ top\_refs}
        \STATE $process\_score \gets \text{$p.process\_score$ + $ref.content\_quality\_score \times \min(2 \times ref.source\_quality\_score, 1.0)$}
    \ENDFOR

    \IF{$all\_reviewed$ \AND $distinctDomainsRequired$ \AND \text{len($top\_refs)$} \geq \text{$O$.$minimumProcessReferences$}$}
        \STATE $p.process\_score \gets \text{$process\_score$} / len(top\_refs)|$
    \ELSE
        \STATE $p.process\_score \gets \text{None}$
    \ENDIF

    \STATE $p.references \gets refReplacements$
    \IF{$p.process\_score = \text{None}$ \OR $p.process\_score$ < $O$.$score\_threshold$ \AND $S$.$reviewIterations$ < $O$.$maximum\_review\_iterations$} 
        \STATE Continue Process Generation Loop
    \ENDIF
    
\ENDFOR
\end{algorithmic}
\end{algorithm}

\begin{algorithm}
\caption{PurgeProcesses}
\label{alg:purge-processes}
\begin{algorithmic}[1]
\REQUIRE Current state $S$, options $O$ (from state)
\FOR{each $p$ in $\text{$S$.$processes$}$}
    \IF{$p$ has insufficient references OR $p.process\_score < O.score\_threshold$ OR $p.scale < O.scale\_threshold$}
        \STATE Remove $p$ from $S$.$processes$
    \ENDIF
\ENDFOR
\end{algorithmic}
\end{algorithm}

\begin{algorithm}
\caption{UpstreamDownstreamExpansion}
\label{alg:expansion}
\begin{algorithmic}[1]
\REQUIRE Current state $S$, options $O$ (from state)
\FOR{each process $p \in S.processes$}
    \STATE $replacement\_precursors$ \gets \emptyset
    \FOR{each precursor $m \in p.precursors$ }
        \IF{$m.hs\_code$ in \text{$S.materials$}}
            \STATE $replacement\_precursors$ \gets replacement\_precursors$ \cup m$
        \ELSE
            \STATE $R$ \gets \CALL{MaterialFinderLLMAgent}{($m$,$s$)}
            \STATE $S.materials$ \gets S.materials$ \cup R$
            \STATE $replacement\_precursors$ \gets replacement\_precursors$ \cup R$
        \ENDIF
    \ENDFOR
    \STATE $replacement\_products$ \gets \emptyset
    \FOR{each product $m \in p.products$ }
        \IF{$m.hs\_code$ in \text{$S.materials$}}
            \STATE $replacement\_products$ \gets replacement\_products$ \cup m$
        \ELSE
            \STATE $R$ \gets \CALL{MaterialFinderLLMAgent}{($m$,$s$)}
            \STATE $S.materials$ \gets S.materials$ \cup R$
            \STATE $replacement\_productss$ \gets replacement\_precursors$ \cup R$
        \ENDIF
    \ENDFOR
\ENDFOR
\end{algorithmic}
\end{algorithm}
